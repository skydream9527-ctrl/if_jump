## 一、关卡配置文件结构

### 1.1 整体结构

```
levels/
├── chapter_01/                    # 第一章：早餐篇
│   ├── config.json                # 章节配置
│   ├── level_01.json              # 1-1关卡
│   ├── level_02.json
│   └── ...
├── chapter_02/                    # 第二章：午餐篇
│   └── ...
└── chapter_10/                    # 第十章：终极篇
    └── ...
```

### 1.2 配置层级

```
GameConfig
├── meta                          # 游戏元信息
│   ├── version
│   ├── lastUpdate
│   └── totalLevels
├── chapters[]                    # 章节数组
│   ├── chapterId
│   ├── name
│   ├── description
│   ├── theme
│   ├── levels[]                  # 关卡数组
│   │   ├── levelId
│   │   ├── name
│   │   ├── platforms[]           # 平台数组
│   │   ├── mechanics[]           # 机制配置
│   │   └── settings              # 关卡设置
│   └── unlockCondition
└── globalSettings                # 全局设置
```

## 四、关卡配置生成工具

### 4.1 关卡生成器配置

```javascript
// LevelGenerator.js
class LevelGenerator {
  constructor(config) {
    this.config = config;
    this.platforms = [];
    this.difficultyCurve = config.difficultyCurve;
  }
  
  generate() {
    // 生成平台
    for (let i = 0; i < this.config.platformCount; i++) {
      const platform = this.generatePlatform(i);
      this.platforms.push(platform);
    }
    
    // 添加机制
    this.addMechanics();
    
    // 添加障碍物
    this.addObstacles();
    
    // 添加道具
    this.addItems();
    
    return this.exportConfig();
  }
  
  generatePlatform(index) {
    const progress = index / this.config.platformCount;
    const difficulty = this.calculateDifficulty(progress);
    
    return {
      id: index + 1,
      type: this.selectPlatformType(difficulty),
      position: this.calculatePosition(index),
      ...this.generatePlatformProperties(difficulty)
    };
  }
  
  calculateDifficulty(progress) {
    // 难度曲线计算
    const base = this.config.baseDifficulty;
    const curve = this.difficultyCurve;
    return base + (1 - base) * Math.pow(progress, curve);
  }
}
```

## 六、热更新配置

### 6.1 关卡数据热更新

```javascript
// LevelHotUpdate.js
class LevelHotUpdate {
  async checkUpdate() {
    const localVersion = await this.getLocalVersion();
    const remoteVersion = await this.fetchRemoteVersion();
    
    if (remoteVersion > localVersion) {
      await this.downloadUpdate(remoteVersion);
      await this.applyUpdate();
    }
  }
  
  async downloadUpdate(version) {
    // 下载新关卡配置
    const configs = await fetch(`/api/levels?v=${version}`);
    // 保存到本地
    await this.saveConfigs(configs);
  }
}
```
